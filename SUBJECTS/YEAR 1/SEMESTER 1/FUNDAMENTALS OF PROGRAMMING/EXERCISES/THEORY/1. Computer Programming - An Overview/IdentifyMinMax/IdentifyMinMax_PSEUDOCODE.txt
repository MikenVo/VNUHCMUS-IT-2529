START
    int a, b, c;
    int max, min;

    INPUT a, b, c

    max = a; min = a;

    IF (min > b) {
        min = b
        IF (min > c) {
            min = c
        }
    }
    ELSE IF (min > c) {
        min = c
    }

    IF (max < b) {
        max = b
        IF (max < c) {
            max = c
        }
    }
    ELSE IF (max < c) {
        max = c
    }

    PRINT max, min
END
