START
    int studentid;

    WHILE (TRUE) {
        INPUT studentid

        IF (TO_STRING(studentid).length() == 8) {
            int sequence, programtype, faculty, admissionyear;

            sequence = studentid % 1000
            studentid /= 1000

            programtype = studentid % 10
            studentid /= 10

            faculty = studentid % 100
            studentid /= 100

            admissionyear = studentid % 100

            PRINT sequence, programtype, faculty, admissionyear
            BREAK;
        }
        ELSE {
            PRINT "ERROR"
            CONTINUE;
        }
    }
END
