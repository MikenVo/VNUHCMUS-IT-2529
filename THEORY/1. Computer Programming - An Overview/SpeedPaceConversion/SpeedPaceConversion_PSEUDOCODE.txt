START
    char answer;

    INPUT answer

    IF (int(answer) == 49) {
        float speed;

        WHILE (TRUE) {
            INPUT speed
            
            IF (speed > 0) {
                PRINT 60/speed
                BREAK
            }
            ELSE {
                PRINT "ERROR"
                CONTINUE
            }
        }
    }
    ELSE {
        float pace;

        WHILE (TRUE) {
            INPUT pace

            IF (pace > 0) {
                PRINT 60/pace
                BREAK
            }
            ELSE {
                PRINT "ERROR"
                CONTINUE
            }
        }
    }
END
